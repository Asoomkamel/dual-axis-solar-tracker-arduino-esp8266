#include <Servo.h> // Corrected include statement for the Servo library

// Constants for servo limits
const int SERVOH_MAX = 180;
const int SERVOH_MIN = 65;
const int SERVOV_MAX = 80;
const int SERVOV_MIN = 15;

// Servo objects
Servo horizontal; // Horizontal servo
Servo vertical;   // Vertical servo

// Initial servo positions
int servohPos = 180;
int servovPos = 45;

// LDR pin assignments
const int LDR_LT = 0; // LDR top left (bottom left on the physical device)
const int LDR_RT = 1; // LDR top right (bottom right on the physical device)
const int LDR_LD = 2; // LDR bottom left (top left on the physical device)
const int LDR_RD = 3; // LDR bottom right (top right on the physical device)

// Tolerance and delay time
const int tolerance = 50;
const int dtime = 10;

void setup() {
  Serial.begin(9600);
  
  // Attach servos to pins
  horizontal.attach(10);
  vertical.attach(8);
  
  // Move to initial positions
  horizontal.write(servohPos);
  vertical.write(servovPos);
  
  // Allow servos to reach the position
  delay(3000);
}

void loop() {
  // Read LDR values
  int lt = analogRead(LDR_LT);
  int rt = analogRead(LDR_RT);
  int ld = analogRead(LDR_LD);
  int rd = analogRead(LDR_RD);

  // Calculate averages
  int avgTop = (lt + rt) / 2;
  int avgBottom = (ld + rd) / 2;
  int avgLeft = (lt + ld) / 2;
  int avgRight = (rt + rd) / 2;

  // Calculate differences
  int diffVert = avgTop - avgBottom;
  int diffHoriz = avgLeft - avgRight;

  // Adjust vertical servo position
  if (abs(diffVert) > tolerance) {
    adjustVertical(diffVert);
  }

  // Adjust horizontal servo position
  if (abs(diffHoriz) > tolerance) {
    adjustHorizontal(diffHoriz);
  }

  // Print values for debugging
  printValues(avgTop, avgBottom, avgLeft, avgRight);

  // Wait before next reading
  delay(dtime);
}

void adjustVertical(int diff) {
  if (diff > 0) {
    servovPos++;
  } else {
    servovPos--;
  }
  servovPos = constrain(servovPos, SERVOV_MIN, SERVOV_MAX);
  vertical.write(servovPos);
}

void adjustHorizontal(int diff) {
  if (diff > 0) {
    servohPos--;
  } else {
    servohPos++;
  }
  servohPos = constrain(servohPos, SERVOH_MIN, SERVOH_MAX);
  horizontal.write(servohPos);
}

void printValues(int top, int bottom, int left, int right) {
  Serial.print(top);
  Serial.print(" ");
  Serial.print(bottom);
  Serial.print(" ");
  Serial.print(left);
  Serial.print(" ");
  Serial.print(right);
  Serial.print(" ");
  Serial.print(dtime);
  Serial.print(" ");
  Serial.println(tolerance);
}
